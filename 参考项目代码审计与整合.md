## 二、六大核心机制分类

审计后发现，所有项目本质上只用了 **6 种去水印手法**：

| 机制 | 适用 | 原理 | 代表项目 |
|------|------|------|----------|
| **M1 原图直链提取** | 图片 | 平台接口/SSR 自带 `image_ori_raw.url`（真·原图，无水印） | #2 #4 #5 #7 #10 #11 #13 #14 |
| **M2 双图合并** | 图片 | `previewImage`(右下干净)+`downloadImage`(左上干净) 拼合 | #4 |
| **M3 视频 lr 参数改写** | 视频 | `get_play_info` 取 `main_url`，把 `lr=video_gen_watermark` → `lr=video_gen_no_watermark` | #3 #5 #6 #9 #10 #11 |
| **M4 网络层 URL 重写** | 图片 | `declarativeNetRequest` 剥离 URL 中的 `-watermark` 片段 | #8 |
| **M5 算法级消除** | 图片(已生成) | inpaint / 扩散重绘 / SynthID 去除 | #15 #16 |
| **M6 15 秒时长解锁** | 视频生成 | 改请求体 `duration=15` 或覆盖 `skill/pack` 响应 | #5 #6 #9 |

> **关键共识**：豆包系图片去水印的"真答案"永远是 `image_ori_raw.url`——服务端原生原图，无水印。
> 视频去水印的"真答案"永远是 `get_play_info` 返回的 `original_media_info.main_url` 且强制 `lr=video_gen_no_watermark`。
> 其它手法（Canvas 重绘、双图合并）都是**取不到原图时的降级/伪去水印方案**，质量有损。

## 三、整合完整代码（带逐段标注）

下面把上述机制整合为一份**统一的浏览器端参考实现**（用户脚本 / 扩展 content 脚本通用结构），
每一段都标注了「来自哪个项目」「做什么」「注意事项」。

```javascript
/* =====================================================================
 * 豆包去水印 · 统一整合参考实现
 * 整合自参考项目 18 个方案，按机制分模块标注
 * 运行环境：浏览器 MAIN world（可访问 window._ROUTER_DATA / 劫持 fetch）
 * ===================================================================== */

/* ---------------------------------------------------------------------
 * 【模块 0】基础设施：消息桥接 (来自 #5 #6 #9)
 * MAIN world 拿到数据 → 通过 window.postMessage 发给扩展世界(ISOLATED)
 * → 扩展世界用 chrome.downloads / chrome.storage 真正下载
 * ------------------------------------------------------------------- */
const BRIDGE = {
  // 页面 → 扩展
  toExt: (type, payload) =>
    window.postMessage({ __ext: true, type, payload }, '*'),
  // 扩展 → 页面（监听回复）
  onExt: (type, cb) =>
    window.addEventListener('message', (e) => {
      if (e.data?.__ext && e.data.type === type) cb(e.data.payload);
    }),
};

/* ---------------------------------------------------------------------
 * 【模块 1】M1 原图直链提取 — 图片去水印核心 (来自 #2 #4 #5 #7 #10 #11)
 * 思路：豆包对话接口/SSE/SSR 中天然返回 image_ori_raw.url（原图，无水印）
 * 我们只需在数据流动时把它捕获下来。
 * ------------------------------------------------------------------- */

// 1a. 劫持 fetch（MAIN world，页面发了什么请求、回了什么我们都能看）
const _origFetch = window.fetch.bind(window);
window.fetch = async (input, init = {}) => {
  const resp = await _origFetch(input, init);
  const url = typeof input === 'string' ? input : input.url;

  // 对话流式接口：从 SSE / JSON 里捞原图 url
  if (url.includes('/chat/completion') || url.includes('/im/chain/single')) {
    const cloned = resp.clone();
    cloned.text().then((text) => collectImageOriRaw(text));
  }
  return resp;
};

// 1b. 解析响应文本，递归找出 image_ori_raw.url（来自 #10 的 JSON.parse 覆写思路）
function collectImageOriRaw(text) {
  try {
    const data = JSON.parse(text);
    deepScan(data, (node) => {
      // 豆包图片原图字段：image_ori_raw.url（真·无水印）
      if (node?.image_ori_raw?.url) {
        BRIDGE.toExt('IMAGE_FOUND', {
          url: node.image_ori_raw.url,
          thumb: node.watermark_url || node.image_thumb?.url, // 仅用于界面预览
        });
      }
    });
  } catch { /* SSE 流不是标准 JSON，逐行解析见模块 1c */ }
}

// 1c. SSE 流逐行解析（来自 #4 #14）：data: 开头的行里藏着 creation
function parseSSE(text) {
  text.split('\n').forEach((line) => {
    if (!line.startsWith('data:')) return;
    const json = line.slice(5).trim();
    const data = safeParse(json);
    // patch_op / event_data 两种格式都可能出现原图
    deepScan(data, (n) => {
      if (n?.image_ori_raw?.url)
        BRIDGE.toExt('IMAGE_FOUND', { url: n.image_ori_raw.url });
    });
  });
}

// 1d. React Fiber 遍历（来自 #4 的黑科技）：即使没拦截到 API，
// 也能从 canvas/img 元素向上翻 React 内部属性找到 previewImage/downloadImage
function getImageInfoFromElement(el) {
  const fiberKey = Object.keys(el).find(
    (k) => k.startsWith('__reactFiber') || k.startsWith('__reactInternalInstance')
  );
  let fiber = el[fiberKey];
  for (let i = 0; i < 45 && fiber; i++, fiber = fiber.return) {
    const props = fiber.memoizedProps || fiber.pendingProps;
    const found = deepScan(props, (n) => n?.image_ori_raw?.url && n);
    if (found) return { url: found.image_ori_raw.url };
  }
  return null;
}

// 1e. SSR 页面数据兜底（来自 #11 #13）：页面 HTML 内联的 _ROUTER_DATA
function collectFromSSR() {
  const m = document.documentElement.innerHTML.match(
    /window\._ROUTER_DATA\s*=\s*(\{.*?\});/s
  );
  if (m) {
    const data = safeParse(m[1]);
    deepScan(data, (n) => {
      if (n?.image_ori_raw?.url)
        BRIDGE.toExt('IMAGE_FOUND', { url: n.image_ori_raw.url });
    });
  }
}

/* ---------------------------------------------------------------------
 * 【模块 2】M2 双图合并 — 取不到原图时的降级方案 (来自 #4)
 * 原理：previewImage 水印在右下水印、downloadImage 水印在左上
 * 把两张"互补干净"的图拼起来，得到完整无水印图。
 * 注意：这是"伪去水印"，质量不如 M1 原图直链。
 * ------------------------------------------------------------------- */
async function mergeWatermark(imgPre, imgDld) {
  const canvas = document.createElement('canvas');
  canvas.width = imgPre.width;
  canvas.height = imgPre.height;
  const ctx = canvas.getContext('2d');
  ctx.drawImage(imgPre, 0, 0);                       // 底图
  const halfW = imgPre.width / 2, halfH = imgPre.height / 2;
  ctx.clearRect(0, 0, halfW, halfH);                // 擦掉左上角水印
  ctx.drawImage(imgDld, 0, 0, halfW, halfH, 0, 0, halfW, halfH); // 用干净左上覆盖
  return canvas.toBlob();
}

/* ---------------------------------------------------------------------
 * 【模块 3】M3 视频 lr 参数改写 + get_play_info (来自 #3 #5 #6 #9 #10 #11)
 * 核心：拿到 vid → 调 get_play_info → 取 original_media_info.main_url
 * → 把 lr=video_gen_watermark 改成 lr=video_gen_no_watermark
 * ------------------------------------------------------------------- */

// 3a. 从分享页 URL 取 video_id（来自 #3）
function getVidFromUrl() {
  return new URLSearchParams(location.search).get('video_id');
}

// 3b. 调豆包 get_play_info（来自 #3 #7 #11）
async function getPlayInfo(vid) {
  const resp = await fetch('https://www.doubao.com/samantha/media/get_play_info', {
    method: 'POST',
    credentials: 'include', // 需要登录态
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ key: vid }),
  });
  const data = await resp.json();
  // original_media_info.main_url 才是原生无水印版本
  return data?.data?.original_media_info?.main_url;
}

// 3c. 强制无水印（来自 #6 #9 #10 的 lr 改写）
function forceNoWatermark(url) {
  return url.replace(/lr=[^&]*/, 'lr=video_gen_no_watermark');
}

// 3d. 完整视频去水印流程
async function downloadVideoNoMark(vid) {
  const raw = await getPlayInfo(vid);
  const clean = forceNoWatermark(raw);
  BRIDGE.toExt('VIDEO_FOUND', { url: clean }); // 交给扩展下载
}

/* ---------------------------------------------------------------------
 * 【模块 4】M4 网络层 URL 重写（来自 #8，唯一不用 JS 劫持的方案）
 * 用 declarativeNetRequest 在请求发出前就把图片 URL 里的 -watermark 剥掉，
 * CDN 直接返回无水印原图。零运行时开销，最干净。
 * 这是写在 manifest.json 的 rules 里，不是 JS 代码：
 *
 * {
 *   "id": 1,
 *   "priority": 1,
 *   "action": {
 *     "type": "redirect",
 *     "redirect": {
 *       "regexSubstitution": "\\1\\2"
 *     }
 *   },
 *   "condition": {
 *     "regexFilter": "(.*)-(dark-watermark|watermark|image-watermark)(.*)",
 *     "resourceTypes": ["image"]
 *   }
 * }
 * ------------------------------------------------------------------- */

/* ---------------------------------------------------------------------
 * 【模块 5】M5 算法级去水印（来自 #15 #16，性质完全不同）
 * 这不是"抓链接"，而是对已经生成/下载下来的图做像素修复。
 * 需在服务端/本地跑 Python（remove-ai-watermarks 库 / ComfyUI 节点）。
 * 这里只列出调用约定，真实算力在 Python 侧：
 *
 *   visible  : 识别已知水印(Gemini/豆包/即梦/Samsung) → inpaint 修复
 *   invisible: SynthID 等隐形水印 → SDXL/ControlNet 扩散重绘
 *   erase    : 指定 MASK 区域擦除 (cv2 telea/ns 或 LaMa)
 *   metadata : 剥离图片元数据中的水印痕迹
 * ------------------------------------------------------------------- */

/* ---------------------------------------------------------------------
 * 【模块 6】M6 15 秒时长解锁（来自 #5 #6 #9）
 * 两种手法互补：
 *   (a) 页面级：劫持 chat/completion 请求体，把 duration 改成 15（来自 #6 #9）
 *   (b) 网络级：用 debugger/Fetch 覆盖 samantha/skill/pack 响应，
 *       在 option_list 里补 5/10/15 秒选项（来自 #5 的 skill-pack JSON）
 * ------------------------------------------------------------------- */

// 6a. 页面级 patch（MAIN world 覆写 fetch 请求体）
const _fetch = window.fetch;
window.fetch = async (input, init = {}) => {
  if (typeof input === 'string' && input.includes('/chat/completion')) {
    const body = JSON.parse(init.body);
    // 把视频生成时长强制设为 15 秒
    if (body?.chat_ability?.ability_param)
      body.chat_ability.ability_param.duration = 15;
    init.body = JSON.stringify(body);
  }
  return _fetch(input, init);
};

// 6b. 网络级 skill-pack 覆盖（service worker 侧，来自 #5）
//   监听 Fetch.requestPaused → 命中 samantha/skill/pack →
//   用本地 skill-pack-response.json 的 option_list（含 15s）fulfillRequest 回去

/* ---------------------------------------------------------------------
 * 【模块 7】UI 注入：下载按钮 + 悬浮面板（来自 #4 #7 #8）
 * 在图片/视频容器上挂按钮，点击触发对应下载。
 * ------------------------------------------------------------------- */
function injectButtons() {
  const observer = new MutationObserver(() => {
    document.querySelectorAll('img, video, .block-video').forEach((el) => {
      if (el.dataset.__marked) return;
      el.dataset.__marked = '1';
      const btn = document.createElement('button');
      btn.textContent = '⬇ 无水印';
      btn.onclick = () => {
        if (el.tagName === 'IMG') {
          // 优先用 M1 捕获的 image_ori_raw；没有则 M2 合并；再没有则原图直下
          const url = window.__oriRawMap?.get(el.src) || el.src;
          BRIDGE.toExt('DOWNLOAD_IMAGE', { url });
        } else {
          const vid = getVidFromElement(el);
          downloadVideoNoMark(vid);
        }
      };
      el.parentElement?.appendChild(btn);
    });
  });
  observer.observe(document.body, { childList: true, subtree: true });
}

/* ---------------------------------------------------------------------
 * 工具函数：深度遍历对象找字段（各项目通用）
 * ------------------------------------------------------------------- */
function deepScan(obj, cb, seen = new WeakSet()) {
  if (!obj || typeof obj !== 'object' || seen.has(obj)) return;
  seen.add(obj);
  cb(obj);
  for (const v of Object.values(obj)) deepScan(v, cb, seen);
}
function safeParse(s) { try { return JSON.parse(s); } catch { return null; } }
```

## 四、各项目"独特亮点"速查（避免重复造轮子）

| 亮点 | 谁做的 | 可借鉴点 |
|------|--------|----------|
| 纯前端 Canvas 取预览图 | #1 | 最简单，适合没有登录态时 |
| **`image_ori_raw` 原图直链** | #2 #4 #5 #7 #10 #11 | **图片去水印唯一正解** |
| React Fiber 遍历兜底 | #4 | 不依赖 API 也能拿图，反爬最稳 |
| 纯同步 ZIP 打包器（不依赖 JSZip） | #4 | 避免豆包页面 polyfill 冲突 |
| debugger + Fetch 覆盖 skill-pack | #5 | 15 秒解锁的"服务端保险"手法 |
| **fplay + qAAB AES-CBC 解密** | #5 | 真·无水印视频直链的进阶获取 |
| MAIN world + forwarder 双世界桥接 | #5 #6 #9 | 既能在页面劫持又安全用扩展 API |
| **declarativeNetRequest URL 重写** | #8 | 零开销、最干净、不读响应 |
| 覆写 `JSON.parse` 拦截全部 JSON | #10 | 比 hook fetch 更底层、更全 |
| 算法级消除（SynthID/可见水印） | #15 #16 | 对已下载图做像素级修复 |

## 五、对我们 V21 插件的启示（关键结论）

> 我们自己的 `豆包视频去水印` V21 插件已经正确采用了最优路径：
> - **图片**：`inject.js`(MAIN) 在 fetch/XHR hook 里直接取 `image_ori_raw.url` ✅（对应 M1，最正解）
> - **视频**：`fallback_api` + `logo_type=unwatermarked` 解码 `main_url` 拿真无水印直链 ✅（比 M3 的 lr 改写更彻底，对应 #5 的 fplay/qAAB 思路）
> - **不用 debugger**：参考项目 #5 证明 debugger 拦截流式接口会破坏豆包对话，我们已规避 ✅
>
> **下一步可借鉴的增强点**：
> 1. 借鉴 #4 的 React Fiber 遍历做 M1 的兜底（防 API 拦截遗漏）
> 2. 借鉴 #8 的 declarativeNetRequest 做图片 URL 重写的"双保险"
> 3. 借鉴 #10 覆写 JSON.parse 而非仅 hook fetch，覆盖更全
> 4. 分享页视频可借鉴 #3 的 `get_play_info` + `video_id` 链路（目前 V21 分享页机制待验证）
