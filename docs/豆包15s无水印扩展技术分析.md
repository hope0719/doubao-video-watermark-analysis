#### 密钥派生
```javascript
async function deriveQaabKeyIv(keySeedText) {
  var keySeedBytes = base64ToBytesLoose(keySeedText);
  var seed32 = keySeedBytes.slice(0, 32);
  var digest1 = await sha512Bytes(seed32);
  var digest2 = await sha512Bytes(concatUint8Arrays([digest1, QAAB_SALT_BYTES]));
  return {
    key: digest2.slice(0, 16),  // AES-128密钥
    iv: digest2.slice(16, 32)   // 初始化向量
  };
}
```

#### 解密过程
```javascript
async function decodeQaabToken(token, keySeedText) {
  var data = base64ToBytesLoose(token);           // Base64解码
  var derived = await deriveQaabKeyIv(keySeedText); // 密钥派生
  var plainBytes = await decryptAesCbc(data, derived.key, derived.iv); // AES-CBC解密
  var unpadded = stripPkcs7(plainBytes);          // PKCS7去填充
  return bytesToMaybeText(unpadded);              // 转为文本
}
```

```javascript
function mutateFplayUrl(fplayUrl, variant) {
  var parsed = new URL(fplayUrl);
  parsed.searchParams.set("codec_type", variant.codecType);  // 编码器类型
  parsed.searchParams.set("logo_type", variant.logoType);    // 水印类型
  return parsed.toString();
}
```

### 3. 无水印验证算法 ✅

```javascript
function isTrueNoWatermarkMediaUrl(url) {
  var parsed = new URL(url);
  var lr = parsed.searchParams.get("lr") || "";
  var cs = parsed.searchParams.get("cs") || "";
  var qs = parsed.searchParams.get("qs") || "";
  
  // 关键条件：cs=0(无水印) + qs=13(高质量) + lr为unwatermarked或无lr
  return cs === "0" && qs === "13" && (!lr || lr === "unwatermarked" || lr === "no_watermark");
}
```

### 4. 评分选择算法 🏆

```javascript
function resolvedCandidateScore(candidate) {
  var score = 0;
  if (candidate.trueNoWatermark) score += 10000;     // 真无水印最高分
  if (candidate.logoType === "unwatermarked") score += 2000;
  if (candidate.logoType === "no_watermark") score += 1500;
  if (candidate.qs === "13") score += 300;            // 高质量参数
  if (candidate.cs === "0") score += 300;             // 无水印参数
  score += candidate.bitrate || 0;                    // 码率加分
  return score;
}
```
